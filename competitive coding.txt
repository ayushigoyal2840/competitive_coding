Q1)count number of set bits between given range of numbers
sol)
#include<bits/stdc++.h>
#define ll long long int
using namespace std;
int countset(int n)
{
	int count =0;
	while(n>0)
	{
		count+=(n&1);
		n=n>>1;
	}
	return count;
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);
	ll t;
	cin>>t;
	while(t--)
	{
		int a,b;
		cin>>a>>b;
		int count=0;
		for(int i=a;i<=b;i++)
		{
			if((i&1)==0 && i<b)
			{
				count+=2*countset(i)+1;
				i++;
				continue;
			}
			count+=countset(i);
		}
		cout<<count;
		}	
}
